<Note>
  Mesosphere SDKs are HTTP-only and connect to the backend on port `8000`.
</Note>

## Python

```python
from mesosphere import HttpClient

client = HttpClient(
    api_url="http://localhost:8000",
    api_key="local-dev-key",
)

client.schema.apply(
    {
        "tables": {
            "users": {
                "fields": {
                    "name": {"type": "string"},
                    "email": {"type": "string"},
                }
            }
        }
    }
)

user_id = client.relational("users").insert({"name": "Theo", "email": "theo@example.com"})
rows = client.relational("users").query(orderBy=[{"field": "name", "direction": "asc"}])
print(user_id, rows)
```

## TypeScript

```ts
import { httpClient } from "mesosphere";

const client = httpClient({
  api_url: "http://localhost:8000",
  api_key: "local-dev-key",
});

await client.schema.apply({
  tables: {
    users: {
      fields: {
        name: { type: "string" },
        email: { type: "string" },
      },
    },
  },
});

const userId = await client.relational("users").insert({
  name: "Theo",
  email: "theo@example.com",
});

const rows = await client.relational("users").query({
  orderBy: [{ field: "name", direction: "asc" }],
});

console.log(userId, rows);
```
